import pandas as pd
import matplotlib.pyplot as plt

# Loading the dataset (downloaded from Lens.org)
file_path = '/Users/bacciogalletti/Desktop/Thesis material/patents-5years.csv'
df = pd.read_csv(file_path)

# Displaying the number of entries (patents)
num_entries = df.shape[0]
print(f"Number of entries (patents): {num_entries}")

# Checking for missing values
missing_values = df.isnull().sum()
print("Missing values per column:")
print(missing_values)

# Displaying the years with the most patents
if 'Publication Date' in df.columns:
    df['Year'] = pd.to_datetime(df['Publication Date'], errors='coerce').dt.year
    patents_per_year = df['Year'].value_counts().sort_index()
    print("\nPatents per year:")
    print(patents_per_year)
    
    # Plotting the number of patents per year
    patents_per_year.plot(kind='bar', title="Patents per Year", xlabel='Year', ylabel='Number of Patents', figsize=(10,6))
    plt.show()

# Displaying patents by type
if 'PatentType' in df.columns:
    patents_per_type = df['PatentType'].value_counts()
    print("\nPatents by type:")
    print(patents_per_type)
    
    # Plotting patents by type
    patents_per_type.plot(kind='pie', autopct='%1.1f%%', title="Patents by Type", figsize=(8,8))
    plt.show()

# Basic analysis: Simple Family Size distribution
if 'Simple Family Size' in df.columns:
    simple_family_size = df['Simple Family Size'].value_counts().sort_index()
    print("\nSimple Family Size distribution:")
    print(simple_family_size)
    
    # Plotting Simple Family Size distribution
    simple_family_size.plot(kind='bar', title="Simple Family Size Distribution", xlabel='Family Size', ylabel='Number of Patents', figsize=(10,6))
    plt.show()

# Other Example: Extended Family Size distribution
if 'Extended Family Size' in df.columns:
    extended_family_size = df['Extended Family Size'].value_counts().sort_index()
    print("\nExtended Family Size distribution:")
    print(extended_family_size)
    
    # Plot Extended Family Size distribution
    extended_family_size.plot(kind='bar', title="Extended Family Size Distribution", xlabel='Family Size', ylabel='Number of Patents', figsize=(10,6))
    plt.show()

# Removing rows where the 'Abstract' column is missing
df = df.dropna(subset=['Abstract'])

import pandas as pd
import nltk
from nltk.corpus import stopwords
import string
import spacy
import re
from sklearn.feature_extraction.text import TfidfVectorizer

# Load English tokenizer, POS tagger, parser, NER, and word vectors
nlp = spacy.load('en_core_web_sm')

# Downloading stopwords for NLTK
nltk.download('stopwords')

# Loading the patent dataset
file_path = '/Users/bacciogalletti/Desktop/Thesis material/2024/patents_5years_2024.csv'
df = pd.read_csv(file_path)

# Combining Title and Abstract into one column
df['Text'] = df['Title'] + ' ' + df['Abstract'].fillna('')

# Function to clean and lemmatize the text
def preprocess_text(text):
    # Lowercase the text
    text = text.lower()
    
    # Removing punctuation and numbers
    text = re.sub(r'\d+', '', text)
    text = text.translate(str.maketrans('', '', string.punctuation))
    
    # Removing stopwords and lemmatize the text
    stop_words = set(stopwords.words('english'))
    doc = nlp(text)
    
    # Lemmatizing and remove stopwords
    tokens = [token.lemma_ for token in doc if token.text not in stop_words and not token.is_punct]
    return ' '.join(tokens)

# Applying the text cleaning and lemmatization function to the combined text
df['Cleaned_Text'] = df['Text'].apply(preprocess_text)

# Displaying the cleaned and lemmatized text
print(df[['Title', 'Cleaned_Text']].head())

# Step 3: TF-IDF analysis and extracting top keywords

# Use TF-IDF to extract key terms
vectorizer = TfidfVectorizer(max_features=1000)  
tfidf_matrix = vectorizer.fit_transform(df['Cleaned_Text'])

# Creating a DataFrame with TF-IDF values
tfidf_df = pd.DataFrame(tfidf_matrix.toarray(), columns=vectorizer.get_feature_names_out(), index=df.index)

# Get top 10 keywords for each patent 
def get_top_keywords(row, top_n=10):
    sorted_indices = row.argsort()[-top_n:][::-1]  # Get indices of top N values
    return [(vectorizer.get_feature_names_out()[i], row[i]) for i in sorted_indices]

# Associating the top 10 keywords to each patent entry
df['Top_Keywords'] = tfidf_df.apply(lambda row: get_top_keywords(row), axis=1)

# Displaying top keywords for each patent (adding a new column)
print(df[['Title', 'Top_Keywords']].head())

# Saving the updated DataFrame with 'Cleaned_Text' and 'Top_Keywords' columns to a CSV file
output_path = '/Users/bacciogalletti/Desktop/Thesis material/2024/patents_with_keywords_2024.csv'
df.to_csv(output_path, index=False)

print(f"Data saved to {output_path}")

import re

# File path to the Hype Cycle PDF
pdf_path = '/Users/bacciogalletti/Desktop/Thesis material/2024/HypeCycle_2024.pdf'

# Open the PDF document
doc = fitz.open(pdf_path)

# Extract text from each page and concatenate it
full_text = ""
for page_num in range(len(doc)):
    page = doc.load_page(page_num)
    full_text += page.get_text()

# Remove text between "" characters and other formatting particularities
cleaned_text = re.sub(r'.*?\n', '', full_text)  # Remove any text between "" and the next newline
cleaned_text = re.sub(r'Gartner, Inc\..*?\nPage \d+ of \d+', '', cleaned_text, flags=re.DOTALL)  # Remove Gartner info and page numbers

pattern = r'Definition:([\s\S]*?)(Why This Is Important)'
